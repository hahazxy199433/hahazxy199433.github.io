# express入门实战

安装express脚手架
```
npm install express-generator -g
```

创建express项目
启动服务 npm start

或者自己手动创建
```
npm init
```
```
npm install express --save-dev
```

创建 app.js

```javascript
var express = require('express');
var app = express();
app.get('/', function (req, res, next) {
    res.json({
        data: 123
    })
});
app.listen(8081,function () {
    console.log('server is running');
})
```
现在在浏览器打开可以看到 {data: 123}

创建public 文件夹，下面创建 scripts 和 stylesheets 文件夹用来存放 js 和css 文件

安装swig
```
npm install swig --save-dev
```
引入swig
```javascript
var express = require('express');
var app = express();
// ↓↓↓↓↓↓↓↓↓↓↓↓↓新加↓↓↓↓↓↓↓↓↓↓↓↓↓↓↓↓

app.engine('html', swig.renderFile);
// 配置路由
app.get('/', function (req, res, next) {
    res.json({

    // 设置返回状态
```

在index.js中添加内容，表单提交ajax
```javascript
$(function (argument) {
    $('#btnsubmit').click(function name(event) {
        event.preventDefault();
        $.ajax({
            url:'/receive',
            type: 'get',
            dataType: 'json',
            data: {
                username: $('#username').val()
            },
            success: function (data) {
                console.info(data);
            },
            error: function (err) {
                alert('error')
            }
        })
    })
})
```

创建数据库 userinfo 字段 id username

安装 mysql 库
```
npm install mysql --save-dev
```

在app.js 中添加操作数据库
```javascript
var express = require('express');
var mysql      = require('mysql');
var app = express();
app.use(express.static('public'));

// 配合swig
var swig = require('swig');
app.set('view engine', 'html');
app.engine('html', swig.renderFile);
// ↓↓↓↓↓↓↓↓↓↓↓↓↓新加↓↓↓↓↓↓↓↓↓↓↓↓↓↓↓↓
// 配置数据库
var connection = mysql.createConnection({
    host     : 'localhost',
    user     : 'root',
    password : '',
    database : 'test'
  });
   
connection.connect();
// 配置路由
app.get('/', function (req, res, next) {
    res.render('index')
});
app.get('/receive', function (req, res, next) {
    var post = {
        username: req.query.username
    };
    var query = connection.query('INSERT INTO userinfo SET?', post, function (err, result) {
        if (err) {
            console.log(err);
            res.json({
                success: 'no',
                msg: '插入数据失败'
            })
        } else {
            res.json({
                success: 'yes',
                msg: '插入数据成功'
            })
        }
    })
});
// ↑↑↑↑↑↑↑↑↑↑↑↑↑↑新加↑↑↑↑↑↑↑↑↑↑↑↑↑↑↑↑
app.get('*', function (req, res, next) {
    // 设置返回状态
    res.status(404);
    res.end('404');
})
app.use(function (err, req, res, next) {
    // 设置返回状态
    res.status(500);
    res.end('error...')
})
app.listen(8081,function () {
    console.log('server is running');
})
```
在页面中输入用户名提交，返回成功，数据成功插入到数据库
